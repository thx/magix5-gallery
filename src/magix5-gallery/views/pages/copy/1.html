 import Magix5 from 'magix5'; export default Magix5.View.extend({ tmpl: '@:index.html', render() { this.digest(); }, 'copy'(e) { this.digest({ success: true, error: false, }) }, 'copy'(e) { this.digest({ success: false, error: true, }) } })
