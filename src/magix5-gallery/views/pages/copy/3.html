被复制内容需要根据操作动态判断，则可配置 text-fn 传入一个 function 处理
注意：text-fn 暂不支持包含异步处理逻辑

JS Code
 import Magix5 from 'magix5'; export default Magix5.View.extend({ tmpl: '@:index.html', init(options) { }, assign(options) { this.set({ textFn: (trigger) => { return trigger.innerText; } }); }, render() { this.digest(); }, 'copy'(e) { this.digest({ successText: e.text, errorText: '', }) }, 'copy'(e) { this.digest({ successText: '', errorText: e.text, }) } })
