以下示例：
当前页数（{{=page}}），每页显示（{{=size}}）
sizes-change="false"可禁止修改分页数
HTML Code
{{!text1}}  
<mx-pagination
    total="{{=total}}" 
    sizes="{{sizes}}"
    size="{{=size}}" 
    page="{{=page}}"
    mx-change="change()" />

<!-- 禁止切换分页数 -->
<mx-pagination
    sizes-change="false"
    total="{{=total}}" 
    sizes="{{sizes}}"
    size="{{=size}}" 
    page="{{=page}}"
    mx-change="change()" />
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            total: 600,
            sizes: [20, 40, 50],
            page: 1,
            size: 40
        });
    },
    'change<change>' (e) {
        // e.page 当前第几页
        // e.size 每页多少条
        // e.offset 偏移量
        this.updater.digest({
            page: e.page,
            size: e.size
        })
    }
});
